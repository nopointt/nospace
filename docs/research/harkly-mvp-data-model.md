## 1. Database Topology

Harkly MVP runs entirely on Cloudflare. Four storage primitives, six logical namespaces:

```
+------------------+     +-------------------+     +-------------------+
|   AUTH_DB (D1)   |     |    KB_DB (D1)     |     |  VECTORIZE_INDEX  |
|                  |     |                   |     |   (Vectorize)     |
|  user            |     |  projects         |     |                   |
|  session         |     |  sources          |     |  1024-dim vectors |
|  account         |     |  documents        |     |  cosine metric    |
|  verification    |     |  document_chunks  |     |  metadata filter  |
+------------------+     |  document_chunks_ |     +-------------------+
                         |    fts (FTS5)     |
+------------------+     |  project_schemas  |     +-------------------+
|   OAUTH_KV (KV)  |     |  schema_fields    |     |  HARKLY_R2 (R2)   |
|                  |     |  extracted_entitie|     |                   |
|  client:{id}     |     |  extracted_relatio|     |  {projectId}/     |
|  grant:{uid}:{g} |     |  ingest_jobs      |     |    {sourceId}/    |
|  token:{uid}:... |     |  ingest_runs      |     |      {filename}   |
+------------------+     |  work_items       |     +-------------------+
                         |  canvas_frames    |
+------------------+     |  canvas_viewports |
|   AUTH_KV (KV)   |     |  query_stats      |
|                  |     +-------------------+
|  rate:{ip}:{win} |
|  session:{tok}   |     +-------------------+
+------------------+     |  SESSION_KV (KV)  |
                         |                   |
                         |  mcp:session:{id} |
                         +-------------------+
```

### Binding Summary (wrangler.toml)

| Binding name | Type | Purpose |
|---|---|---|
| `AUTH_DB` | D1 | better-auth user accounts, sessions |
| `KB_DB` | D1 | Knowledge base: projects, sources, documents, chunks, schemas, extractions, canvas, jobs |
| `VECTORIZE_INDEX` | Vectorize | Vector similarity search (1024-dim, cosine) |
| `HARKLY_R2` | R2 | Original file storage (PDFs, DOCX, audio, video) |
| `OAUTH_KV` | KV | workers-oauth-provider token storage (OAuth 2.1 grants, client registrations) |
| `AUTH_KV` | KV | better-auth session cache + rate limiting |
| `SESSION_KV` | KV | MCP session state (identity-bound, TTL) |

### Why Two D1 Databases

1. **Isolation**: Auth schema is managed by better-auth CLI migrations. KB schema is managed by Drizzle. Separate migration lifecycles prevent conflicts.
2. **Size**: D1 hard limit is 10 GB per database. Auth data stays small (<100 MB). KB data grows with user content and can approach the limit independently.
3. **Security**: Auth DB contains password hashes and session tokens. KB DB contains user content. Separate blast radius.

## 3. D1 Schema -- Knowledge Base DB (KB_DB)

### 3.1 Core Tables

```sql
-- ============================================================
-- KB_DB: Core document storage
-- Migration: 0001_core_tables.sql
-- ============================================================

-- Projects: top-level research container, scoped to a user
CREATE TABLE IF NOT EXISTS projects (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID
    tenant_id TEXT NOT NULL,                     -- user.id from AUTH_DB (tenant isolation)
    title TEXT NOT NULL,
    description TEXT,
    frame_type TEXT,                             -- 'jtbd' | 'spice' | 'peo' | 'picot' | 'issue_tree' | 'hmw' | 'free_form'
    frame_data TEXT,                             -- JSON: framework-specific structure
    status TEXT NOT NULL DEFAULT 'active',       -- 'active' | 'archived'
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_projects_tenant ON projects(tenant_id);

-- Sources: a data origin (file upload, URL, API connector)
CREATE TABLE IF NOT EXISTS sources (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID
    project_id TEXT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    tenant_id TEXT NOT NULL,
    type TEXT NOT NULL,                         -- 'pdf' | 'docx' | 'csv' | 'txt' | 'url' | 'audio' | 'video' | 'xlsx'
    title TEXT,
    url TEXT,                                   -- original URL if applicable
    r2_key TEXT,                                -- R2 object key for uploaded files
    mime_type TEXT,
    file_size INTEGER,                          -- bytes
    status TEXT NOT NULL DEFAULT 'pending',     -- 'pending' | 'processing' | 'processed' | 'failed'
    error_message TEXT,
    metadata TEXT,                              -- JSON: source-specific metadata
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_sources_project ON sources(project_id);
CREATE INDEX idx_sources_tenant ON sources(tenant_id);

-- Documents: extracted text representation of a source
-- One source may produce one document (PDF -> text) or many (CSV -> rows)
CREATE TABLE IF NOT EXISTS documents (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID
    project_id TEXT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    source_id TEXT NOT NULL REFERENCES sources(id) ON DELETE CASCADE,
    tenant_id TEXT NOT NULL,
    title TEXT,
    content TEXT NOT NULL,                      -- full extracted text (guarded by TokenCounter: max 1.5MB)
    word_count INTEGER,
    language TEXT DEFAULT 'en',
    screening_status TEXT NOT NULL DEFAULT 'pending',  -- 'pending' | 'included' | 'excluded' | 'maybe' | 'flagged'
    screening_reason TEXT,
    relevance_score REAL,
    metadata TEXT,                              -- JSON: language, author, publish_date, etc.
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_documents_project ON documents(project_id);
CREATE INDEX idx_documents_source ON documents(source_id);
CREATE INDEX idx_documents_tenant ON documents(tenant_id);
CREATE INDEX idx_documents_screening ON documents(project_id, screening_status);

-- Document chunks: split text for embedding and retrieval
CREATE TABLE IF NOT EXISTS document_chunks (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID, also used as Vectorize vector ID
    document_id TEXT NOT NULL REFERENCES documents(id) ON DELETE CASCADE,
    project_id TEXT NOT NULL,
    tenant_id TEXT NOT NULL,
    chunk_index INTEGER NOT NULL,               -- order within document (0-based)
    content TEXT NOT NULL,                       -- chunk text (typically 500 chars, max safe: 2000 for Vectorize metadata)
    token_count INTEGER,
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_chunks_document ON document_chunks(document_id);
CREATE INDEX idx_chunks_project ON document_chunks(project_id);
CREATE INDEX idx_chunks_tenant ON document_chunks(tenant_id);

-- FTS5 virtual table for full-text search on chunks
CREATE VIRTUAL TABLE IF NOT EXISTS document_chunks_fts USING fts5(
    id UNINDEXED,
    document_id UNINDEXED,
    project_id UNINDEXED,
    tenant_id UNINDEXED,
    content,
    content = 'document_chunks'
);

-- Trigger: auto-sync FTS5 on INSERT
CREATE TRIGGER IF NOT EXISTS chunks_fts_insert AFTER INSERT ON document_chunks
BEGIN
    INSERT INTO document_chunks_fts(id, document_id, project_id, tenant_id, content)
    VALUES (new.id, new.document_id, new.project_id, new.tenant_id, new.content);
END;

-- Trigger: auto-sync FTS5 on DELETE
CREATE TRIGGER IF NOT EXISTS chunks_fts_delete AFTER DELETE ON document_chunks
BEGIN
    INSERT INTO document_chunks_fts(document_chunks_fts, id, document_id, project_id, tenant_id, content)
    VALUES ('delete', old.id, old.document_id, old.project_id, old.tenant_id, old.content);
END;

-- Trigger: auto-sync FTS5 on UPDATE
CREATE TRIGGER IF NOT EXISTS chunks_fts_update AFTER UPDATE ON document_chunks
BEGIN
    INSERT INTO document_chunks_fts(document_chunks_fts, id, document_id, project_id, tenant_id, content)
    VALUES ('delete', old.id, old.document_id, old.project_id, old.tenant_id, old.content);
    INSERT INTO document_chunks_fts(id, document_id, project_id, tenant_id, content)
    VALUES (new.id, new.document_id, new.project_id, new.tenant_id, new.content);
END;
```

**FTS5 query pattern** (BM25 ranking -- note: rank is negative in SQLite FTS5, lower = better match):
```sql
SELECT c.id, c.document_id, c.content, f.rank
FROM document_chunks_fts f
JOIN document_chunks c ON c.id = f.id
WHERE document_chunks_fts MATCH ?
  AND f.tenant_id = ?
  AND f.project_id = ?
ORDER BY f.rank ASC    -- ASC because BM25 rank is negative in FTS5
LIMIT 20;
```

### 3.2 Schema / Extraction Tables

These tables power Harkly's killer feature: AI proposes schema from data, user confirms, system extracts structured entities.

```sql
-- ============================================================
-- KB_DB: Schema and extraction tables
-- Migration: 0002_schema_extraction.sql
-- ============================================================

-- Project schemas: user-confirmed schema definitions per project
CREATE TABLE IF NOT EXISTS project_schemas (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID
    project_id TEXT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    tenant_id TEXT NOT NULL,
    name TEXT NOT NULL,                         -- e.g. "Interview Analysis", "Contract Clauses"
    version INTEGER NOT NULL DEFAULT 1,
    status TEXT NOT NULL DEFAULT 'draft',       -- 'draft' | 'confirmed' | 'archived'
    discovery_prompt TEXT,                      -- the prompt used for schema discovery
    model_used TEXT,                            -- LLM model that generated this schema
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_schemas_project ON project_schemas(project_id);
CREATE INDEX idx_schemas_tenant ON project_schemas(tenant_id);

-- Schema fields: individual fields within a schema (recursive tree via parent_field_id)
CREATE TABLE IF NOT EXISTS schema_fields (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID
    schema_id TEXT NOT NULL REFERENCES project_schemas(id) ON DELETE CASCADE,
    parent_field_id TEXT REFERENCES schema_fields(id) ON DELETE CASCADE,  -- NULL for root fields
    name TEXT NOT NULL,                         -- field name (snake_case)
    display_name TEXT,                          -- human-readable label
    type TEXT NOT NULL,                         -- 'string' | 'number' | 'boolean' | 'date' | 'array' | 'object' | 'enum'
    description TEXT,                           -- AI-generated or user-edited description
    required INTEGER NOT NULL DEFAULT 0,        -- boolean
    enum_values TEXT,                           -- JSON array of allowed values (for type='enum')
    extraction_hints TEXT,                      -- hints for the extraction LLM
    sort_order INTEGER NOT NULL DEFAULT 0,
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_fields_schema ON schema_fields(schema_id);
CREATE INDEX idx_fields_parent ON schema_fields(parent_field_id);

-- Extracted entities: structured data extracted from documents using confirmed schema
CREATE TABLE IF NOT EXISTS extracted_entities (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID
    project_id TEXT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    schema_id TEXT NOT NULL REFERENCES project_schemas(id) ON DELETE CASCADE,
    document_id TEXT NOT NULL REFERENCES documents(id) ON DELETE CASCADE,
    tenant_id TEXT NOT NULL,
    entity_type TEXT NOT NULL,                  -- matches a root-level schema field name or entity type
    data TEXT NOT NULL,                         -- JSON: extracted key-value pairs conforming to schema
    confidence REAL,                            -- 0.0 to 1.0
    evidence TEXT,                              -- direct quote from source supporting extraction
    verified INTEGER NOT NULL DEFAULT 0,        -- boolean: human verified
    rejected INTEGER NOT NULL DEFAULT 0,        -- boolean: human rejected
    annotation TEXT,                            -- user notes on this extraction
    created_at TEXT NOT NULL DEFAULT (datetime('now')),
    updated_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_entities_project ON extracted_entities(project_id);
CREATE INDEX idx_entities_schema ON extracted_entities(schema_id);
CREATE INDEX idx_entities_document ON extracted_entities(document_id);
CREATE INDEX idx_entities_tenant ON extracted_entities(tenant_id);
CREATE INDEX idx_entities_type ON extracted_entities(project_id, entity_type);

-- Extracted relations: relationships between extracted entities
CREATE TABLE IF NOT EXISTS extracted_relations (
    id TEXT PRIMARY KEY NOT NULL,               -- ULID
    project_id TEXT NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    schema_id TEXT NOT NULL REFERENCES project_schemas(id) ON DELETE CASCADE,
    tenant_id TEXT NOT NULL,
    relation_type TEXT NOT NULL,                -- e.g. 'works_for', 'caused_by', 'mentions'
    source_entity_id TEXT NOT NULL REFERENCES extracted_entities(id) ON DELETE CASCADE,
    target_entity_id TEXT NOT NULL REFERENCES extracted_entities(id) ON DELETE CASCADE,
    confidence REAL,
    evidence TEXT,                              -- quote supporting this relation
    metadata TEXT,                              -- JSON: additional attributes
    created_at TEXT NOT NULL DEFAULT (datetime('now'))
);
CREATE INDEX idx_relations_project ON extracted_relations(project_id);
CREATE INDEX idx_relations_source ON extracted_relations(source_entity_id);
CREATE INDEX idx_relations_target ON extracted_relations(target_entity_id);
CREATE INDEX idx_relations_type ON extracted_relations(project_id, relation_type);
```

**SchemaField intermediate representation** (used between LLM output and DB storage):
```typescript
type SchemaField = {
    name: string;
    type: "string" | "number" | "boolean" | "date" | "array" | "object" | "enum";
    description?: string;
    required?: boolean;
    children?: SchemaField[];     // for array/object types
    enumValues?: string[];        // for enum type
    extractionHints?: string;     // guidance for extraction LLM
};
```

## 4. Vectorize Index Config

### Index Definition

| Property | Value | Rationale |
|---|---|---|
| **Name** | `harkly-kb` (prod) / `harkly-kb-dev` (dev) | Separate prod/dev to prevent pollution |
| **Dimensions** | 1024 | `@cf/baai/bge-large-en-v1.5` output dimension |
| **Distance metric** | cosine | Standard for text embeddings; BGE models are cosine-normalized |
| **Max vectors** | 5,000,000 per index | Vectorize hard limit |

### Vectorize CLI Setup

```bash
# Production
wrangler vectorize create harkly-kb --dimensions=1024 --metric=cosine

# Development
wrangler vectorize create harkly-kb-dev --dimensions=1024 --metric=cosine
```

### wrangler.toml Bindings

```toml
# Production
[[vectorize]]
binding = "VECTORIZE_INDEX"
index_name = "harkly-kb"

# Dev override via wrangler.toml [env.dev] or separate file
```

### Vector Metadata Schema

Each vector stored in Vectorize carries metadata for filtering and retrieval:

```typescript
interface VectorMetadata {
    tenantId: string;       // user ID -- primary isolation filter
    projectId: string;      // project ID -- secondary filter
    documentId: string;     // parent document
    chunkId: string;        // matches document_chunks.id
    chunkIndex: number;     // order within document
    content: string;        // TRUNCATED to 2000 chars -- for quick retrieval without D1 roundtrip
    sourceType: string;     // 'pdf' | 'docx' | etc.
}
```

### Namespace Strategy

**No Vectorize namespaces used.** Isolation is achieved via metadata filtering:

```typescript
const results = await env.VECTORIZE_INDEX.query(embedding, {
    topK: 10,
    returnMetadata: "all",
    filter: {
        tenantId: userId,
        projectId: projectId,
    },
});
```

**Rationale**: Vectorize namespaces are static (cannot be created dynamically per user). Metadata filtering on `tenantId` + `projectId` provides equivalent isolation with dynamic scoping.

### Query Pattern (Hybrid RAG)

```
1. User query arrives
2. Rewrite to 3-5 variant queries via LLM (@cf/meta/llama-3.1-8b-instruct, free)
3. Parallel:
   a. FTS5 MATCH on document_chunks_fts (filtered by project_id + tenant_id)
   b. For each rewritten query:
      - Embed via @cf/baai/bge-large-en-v1.5
      - Vectorize.query(embedding, topK=10, filter={tenantId, projectId})
4. Reciprocal Rank Fusion (RRF) merges FTS + vector results
5. Fetch full chunk text from D1 by merged IDs
6. Inject as context -> stream LLM response
```

### Constraints

- **topK hard cap**: 20 per Vectorize query. With 5 query rewrites, effective candidates = up to 100 (deduplicated).
- **Metadata content**: truncated to 2000 chars. Full chunk text lives in D1 `document_chunks.content`.
- **Vector ID format**: must be alphanumeric + dash/underscore, max 64 chars. Use `IdUtils.ensureSafeId(ulid)`.

### Bucket Name


### Upload Flow (Presigned URL Pattern)

```
1. Client: POST /api/upload/presign { projectId, filename, contentType, fileSize }
2. Server:
   a. Validate: fileSize < 100MB (Worker request body limit)
   b. Generate R2 key: {tenantId}/{projectId}/{sourceId}/{filename}
   c. Create source row in D1 (status='pending', r2_key=key)
   d. Generate presigned PUT URL via aws4fetch (15 min expiry)
   e. Return { presignedUrl, sourceId }
3. Client: PUT presignedUrl with file body (direct to R2, bypasses Worker)
4. Client: POST /api/upload/confirm { sourceId }
5. Server:
   a. Verify R2 object exists at key
   b. Update source.status = 'processing'
   c. Enqueue CF Queue job: { jobType: 'process_source', sourceId }
```

### R2 Lifecycle

- **Write**: on upload via presigned URL
- **Read**: during ingestion (Worker reads file for text extraction)
- **No public read**: files are never served directly to clients. Content is served via D1 (extracted text) or Worker API.
- **Deletion**: when source is deleted, R2 object is deleted via `env.HARKLY_R2.delete(key)`

## 6. KV Namespaces

### OAUTH_KV (workers-oauth-provider)

Managed entirely by `@cloudflare/workers-oauth-provider`. Harkly does not read/write directly.

| Key pattern | Value | TTL |
|---|---|---|
| `client:{clientId}` | Client registration JSON | None (permanent) |
| `grant:{userId}:{grantId}` | Grant + AES-GCM encrypted props | refreshTokenTTL (30 days) |
| `token:{userId}:{grantId}:{tokenId}` | Access token metadata (SHA-256 hash) | accessTokenTTL (1 hour) |

```toml
[[kv_namespaces]]
binding = "OAUTH_KV"
id = "<uuid>"
```

### AUTH_KV (better-auth sessions + rate limiting)

Used by better-auth's `withCloudflare({ kv: env.AUTH_KV })`.

| Key pattern | Value | TTL |
|---|---|---|
| `rate:{ip}:{window}` | Request count (integer as string) | 60s (minimum KV TTL) |
| `session:{token_hash}` | Session data JSON | Session expiry |

```toml
[[kv_namespaces]]
binding = "AUTH_KV"
id = "<uuid>"
```

### SESSION_KV (MCP session state)

Replaces mcp-ts-template's in-memory `Map<string, Session>` with persistent KV.

| Key pattern | Value | TTL |
|---|---|---|
| `mcp:session:{sessionId}` | `{ tenantId, clientId, subject, createdAt, lastAccessed }` | 24 hours |

Identity binding: session is validated against `{tenantId, clientId}` on every request to prevent session hijacking.

```toml
[[kv_namespaces]]
binding = "SESSION_KV"
id = "<uuid>"
```

## 9. Constraints and Guards

### TokenCounter (D1 Row Size Guard)

Adapted from openai-sdk-knowledge-org. Prevents `SQLITE_TOOBIG` errors.

```typescript
// src/lib/guards/token-counter.ts

export const TokenCounter = {
    /** D1 max row size (SQLite page limit) */
    D1_MAX_ROW_SIZE: 2_000_000,        // ~2MB

    /** Safe content size (75% of max to leave room for other columns) */
    D1_SAFE_CONTENT_SIZE: 1_500_000,   // ~1.5MB

    /** Vectorize metadata content truncation limit */
    VECTORIZE_METADATA_LIMIT: 2_000,   // 2000 chars

    /**
     * Validate and truncate content to fit D1 row size limit.
     * Returns truncated content if too large, original otherwise.
     */
    validateAndTruncateContent(content: string): {
        content: string;
        truncated: boolean;
        originalSize: number;
    } {
        const size = new TextEncoder().encode(content).length;
        if (size <= this.D1_SAFE_CONTENT_SIZE) {
            return { content, truncated: false, originalSize: size };
        }
        // Binary search for safe truncation point (UTF-8 aware)
        const decoder = new TextDecoder();
        const encoded = new TextEncoder().encode(content);
        const truncated = decoder.decode(encoded.slice(0, this.D1_SAFE_CONTENT_SIZE));
        return {
            content: truncated,
            truncated: true,
            originalSize: size,
        };
    },

    /**
     * Truncate content for Vectorize metadata field.
     */
    truncateForVectorize(content: string): string {
        if (content.length <= this.VECTORIZE_METADATA_LIMIT) {
            return content;
        }
        return content.substring(0, this.VECTORIZE_METADATA_LIMIT);
    },
};
```

### IdUtils (Vectorize ID Safety)

Vectorize vector IDs must be alphanumeric + dash/underscore, max 64 chars.

```typescript
// src/lib/guards/id-utils.ts

import { ulid } from "ulid";

export const IdUtils = {
    /**
     * Generate a new ULID. Safe for use as D1 primary key and Vectorize vector ID.
     */
    newId(): string {
        return ulid();
    },

    /**
     * Ensure a string is safe for use as a Vectorize vector ID.
     * Strips unsafe characters, truncates to 64 chars.

    ensureSafeId(id: string): string {

    vectorId(chunkId: string): string {
```

### Batch Limits (SQLite Variable Guard)

SQLite has a hard limit of 999 variables in a single statement. All `WHERE id IN (...)` queries must batch.

```typescript
// src/lib/guards/batch.ts

/**
 * Execute a callback in batches of `size` items.
 * Prevents SQLite SQLITE_MAX_VARIABLE_NUMBER (999) errors.
 * Recommended batch size: 100 (conservative, leaves room for other params).
 */
export async function batchProcess<T, R>(
    items: T[],
    size: number,
    fn: (batch: T[]) => Promise<R[]>,
): Promise<R[]> {
    const results: R[] = [];
    for (let i = 0; i < items.length; i += size) {
        const batch = items.slice(i, i + size);
        const batchResults = await fn(batch);
        results.push(...batchResults);
    }
    return results;
}

/**
 * Batch IDs for IN(...) queries. Max 100 per batch.
 */
export function batchIds(ids: string[], maxPerBatch: number = 100): string[][] {
    const batches: string[][] = [];
    for (let i = 0; i < ids.length; i += maxPerBatch) {
        batches.push(ids.slice(i, i + maxPerBatch));
    }
    return batches;
}
```

### Vectorize Insert Rate Limiter

```typescript
// src/lib/guards/rate-limiter.ts

export class RateLimiter {
    private requestCount = 0;
    private windowStart = Date.now();

    constructor(
        private readonly maxRequestsPerMinute: number = 100,
        private readonly delayBetweenBatchesMs: number = 500,
    ) {}

    async waitIfNeeded(): Promise<void> {
        const now = Date.now();
        const elapsed = now - this.windowStart;

        if (elapsed >= 60_000) {
            this.requestCount = 0;
            this.windowStart = now;
        }

        this.requestCount++;

        if (this.requestCount >= this.maxRequestsPerMinute) {
            const waitMs = 60_000 - elapsed + this.jitter();
            await new Promise((resolve) => setTimeout(resolve, waitMs));
            this.requestCount = 0;
            this.windowStart = Date.now();
        }
    }

    async delayBetweenBatches(): Promise<void> {
        await new Promise((resolve) =>
            setTimeout(resolve, this.delayBetweenBatchesMs + this.jitter())
        );
    }

    private jitter(): number {
        return Math.floor(Math.random() * 100);
    }
}
```
