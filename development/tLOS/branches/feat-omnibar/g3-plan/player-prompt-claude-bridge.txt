'use strict';

const { connect, StringCodec } = require('nats');
const { spawn } = require('child_process');

const NATS_URL = process.env.NATS_URL || 'nats://localhost:4222';
const MODEL_NAME = 'claude-sonnet-4-6';
const sc = StringCodec();

// Map: tLOS sessionId -> { claudeSessionId, prevTextLen }
// prevTextLen tracks how much text was already sent (for delta extraction)
const sessions = new Map();

function publish(nc, subject, obj) {
    nc.publish(subject, sc.encode(JSON.stringify(obj)));
}

async function handleChat(nc, sessionId, content) {
    const session = sessions.get(sessionId) || { claudeSessionId: null, prevTextLen: 0 };

    const args = [
        '-p', content,
        '--output-format', 'stream-json',
        '--include-partial-messages'
    ];

    if (session.claudeSessionId) {
        args.push('--resume', session.claudeSessionId);
    }

    // Unset CLAUDECODE to bypass nested session restriction
    const env = Object.assign({}, process.env);
    delete env.CLAUDECODE;

    const child = spawn('claude', args, { env, shell: false });

    let buf = '';
    let currentTextLen = 0; // tracks chars already published for current assistant message

    child.stdout.on('data', (chunk) => {
        buf += chunk.toString();
        const lines = buf.split('\n');
        buf = lines.pop(); // keep incomplete last line

        for (const line of lines) {
            const trimmed = line.trim();
            if (!trimmed) continue;

            let msg;
            try { msg = JSON.parse(trimmed); } catch { continue; }

            // Capture claude session_id for resume (multi-turn)
            if (msg.type === 'system' && msg.subtype === 'init' && msg.session_id) {
                sessions.set(sessionId, { claudeSessionId: msg.session_id, prevTextLen: 0 });
                currentTextLen = 0;
            }

            // Extract text delta from assistant messages
            if (msg.type === 'assistant' && msg.message && msg.message.content) {
                for (const block of msg.message.content) {
                    if (block.type === 'text' && typeof block.text === 'string') {
                        const fullText = block.text;
                        const delta = fullText.slice(currentTextLen);
                        if (delta.length > 0) {
                            currentTextLen = fullText.length;
                            publish(nc, 'tlos.shell.broadcast', {
                                type: 'agent:token',
                                payload: { sessionId, delta, done: false }
                            });
                        }
                    }
                }
            }

            // result = done
            if (msg.type === 'result') {
                // Update stored session
                const stored = sessions.get(sessionId);
                if (stored) stored.prevTextLen = 0;
                currentTextLen = 0;

                publish(nc, 'tlos.shell.broadcast', {
                    type: 'agent:token',
                    payload: { sessionId, delta: '', done: true }
                });
            }
        }
    });

    child.stderr.on('data', (chunk) => {
        const text = chunk.toString().trim();
        if (text) console.error('[claude stderr]', text);
    });

    child.on('error', (err) => {
        console.error('[claude spawn error]', err.message);
        publish(nc, 'tlos.shell.broadcast', {
            type: 'agent:token',
            payload: { sessionId, delta: `[Error: ${err.message}]`, done: true }
        });
    });

    child.on('close', (code) => {
        if (code !== 0) {
            publish(nc, 'tlos.shell.broadcast', {
                type: 'agent:token',
                payload: { sessionId, delta: '', done: true }
            });
        }
    });
}

async function main() {
    console.log('[tlos-claude-bridge] connecting to NATS:', NATS_URL);
    const nc = await connect({ servers: NATS_URL });
    console.log('[tlos-claude-bridge] connected');

    // Publish ready status
    publish(nc, 'tlos.shell.broadcast', {
        type: 'agent:status',
        payload: { ready: true, model: MODEL_NAME }
    });

    console.log(`[tlos-claude-bridge] online — model=${MODEL_NAME}`);

    const sub = nc.subscribe('tlos.shell.events');

    for await (const msg of sub) {
        let parsed;
        try { parsed = JSON.parse(sc.decode(msg.data)); } catch { continue; }

        if (parsed.type !== 'agent:chat') continue;

        const { sessionId, content } = parsed;
        if (!sessionId || !content) continue;

        handleChat(nc, sessionId, content).catch(err => {
            console.error('[handleChat error]', err.message);
        });
    }

    await nc.drain();
}

main().catch(err => {
    console.error('[fatal]', err);
    process.exit(1);
});
