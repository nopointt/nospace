}

// Auth status: read ~/.claude.json for known account/token fields
function getAuthStatus() {
    const claudeJson = path.join(process.env.USERPROFILE || process.env.HOME || '', '.claude.json');
    let authed = false;
    try {
        const data = JSON.parse(fs.readFileSync(claudeJson, 'utf8'));
        authed = !!(
            data.oauthAccount ||
            data.primaryAccountId ||
            data.oauthToken ||
            data.accessToken ||
            data.claudeAiOAuthAccount ||
            (data.accounts && Object.keys(data.accounts).length > 0)
        );
    } catch { /* not found or invalid */ }
    return { found: true, authed, path: CLAUDE_EXE };
}

function publishAuthStatus(nc) {
    publish(nc, 'tlos.shell.broadcast', {
        type: 'agent:auth:status',
        payload: getAuthStatus()
    });
}
