### 3. src/lib/extraction/zod-compiler.ts — Convert SchemaField[] to Zod schema
Import { z } from "zod" (add to package.json if needed, but assume installed)
Actually use: import { z, ZodObject, ZodType } from "zod"

Export function convertToZodSchema(fields: SchemaField[]): z.ZodObject<any>
- Recursive tree-walk:
  - string → z.string().optional() (or .required based on field.required)
  - number → z.number().optional()
  - boolean → z.boolean().optional()
  - date → z.string().optional() (dates as ISO strings)
  - enum → z.enum(field.enumValues as [string, ...string[]]).optional()
  - object → z.object(recurse children).optional()
  - array → z.array(recurse children[0] or z.string()).optional()
- If field.required, don't add .optional()
- Return z.object({...})

Export type SchemaField = {
  name: string;
  type: "string" | "number" | "boolean" | "date" | "array" | "object" | "enum";
  description?: string;
  required?: boolean;
  children?: SchemaField[];
  enumValues?: string[];
  extractionHints?: string;
}
